#include <bits/stdc++.h>
using namespace std;
const int N = 50;
unordered_map <int,int> l,r,pos;
int postorder[N],inorder[N];
int build(int il,int ir,int pl,int pr)
{
    int root = postorder[pr];
    int k = pos[root];
    if(k > il) l[root] = build(il,k-1,pl,pl+k-il-1);
    if(k < ir) r[root] = build(k+1,ir,pl+k-il,pr-1);
    return root;
}
void bfs(int root)
{
    queue <int> q;
    q.push(root);
    while(!q.empty())
    {
        auto t = q.front();
        q.pop();
        cout << t << ' ';
        if(l.count(t)) q.push(l[t]);
        if(r.count(t)) q.push(r[t]);
    }
}
int main()
{
    int n;
    cin >> n;
    for(int i=0;i<n;i++)
        cin >> postorder[i];

    for(int i=0;i<n;i++)
    {
        cin >> inorder[i];
        pos[inorder[i]] = i;
    }
    int root = build(0,n-1,0,n-1);
    bfs(root);
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int N = 100010;
int n;
int a[N];
int s[N];
int main()
{
    cin >> n;
    for(int i=1;i<=n;i++)
       cin >> a[i];
    for(int i=1;i<=n;i++)
        s[i] = s[i-1] + a[i];
    int maxx = -1e9;
    int l = 0 , r = 0;
    for(int i=1;i<=n;i++)
        for(int j=i;j<=n;j++)
        {
            if(maxx < s[j] - s[i - 1])
            {
                maxx = s[j] - s[i - 1];
                l = i , r = j;
            }
        }
    if(maxx >= 0)
        cout << maxx << ' ' << a[l] << ' ' << a[r] << endl;
    else cout << 0 << ' ' << a[1] << ' ' << a[n] << endl;
    return 0;
}

活动打卡代码 AcWing 3995. 最小的和

#include <bits/stdc++.h>
using namespace std;
const int N = 1010;
int a[N],b[N];
int n;
long long ans = 0;
long long cal()
{
    for(int i=1;i<=n;i++)
        ans += pow(a[i] - b[i],2);
    return ans;
}
int find_max()
{
    int max_id = 1;
    int maxx = 0;
    for(int i=1;i<=n;i++)
    {
        int s = abs(a[i] - b[i]);
        if(s > maxx)
        {
            maxx = s;
            max_id = i;
        }
    }
    return max_id;
}
int main()
{
    int k1,k2;
    cin >> n >> k1 >> k2;
    for(int i=1;i<=n;i++)
        cin >> a[i];
    for(int i=1;i<=n;i++)
        cin >> b[i];    

    while(k1--)
    {
        int id = find_max();
        if(a[id] == b[id]) break;
        if(a[id] > b[id]) a[id]--;
        else a[id]++;
    }

    k1++;
    if(k1 % 2 != 0) a[1]++; 

    while(k2--)
    {
        int id = find_max();
        if(a[id] == b[id]) break;
        if(a[id] > b[id]) b[id]++;
        else b[id]--;
    }
    k2++;
    if(k2 % 2 != 0) b[1]++;
    cout << cal() << endl;
    return 0;
}

活动打卡代码 AcWing 1239. 乘积最大
Move.
10天前 
#include <bits/stdc++.h>
using namespace std;
const int N = 100010 , mod = 1000000009;
typedef long long ll;
int n,k;
int a[N];
int main()
{
    cin >> n >> k;
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);

    sort(a,a+n);

    ll res = 1 , l = 0 , r = n - 1;
    int sign = 1;
    if(k % 2)
    {
        res = a[r--];
        k--;
        if(res < 0) sign = -1;
    }

    while(k)
    {
        ll x = (ll) a[l] * a[l+1] , y = (ll) a[r] * a[r-1];
        if(x * sign > y * sign)
        {
            res = x % mod * res % mod;
            l += 2;
        }
        else
        {
            res = y % mod * res % mod;
            r -= 2;
        }
        k -= 2;
    }
    cout << res << endl;
    return 0;
}

#include <bits/stdc++.h>
using namespace std;
const int N = 500010;
int n;
long double s;
int a[N];
int main()
{
    cin >> n >> s;
    for(int i=0;i<n;i++)
        scanf("%d",&a[i]);

    sort(a,a+n);
    long double res = 0 , avg = s / n;
    for(int i=0;i<n;i++)
    {
        double cur = s / (n - i);
        if(a[i] < cur) cur = a[i];
        res += (cur - avg) * (cur - avg);
        s -= cur;
    }
    printf("%.4Lf",sqrt(res / n));
    return 0;
}

活动打卡代码 AcWing 112. 雷达设备
Move.
12天前

把问题转换成区间选点问题

#include <bits/stdc++.h>
using namespace std;
const int N = 1010;
int n,d;
struct Seg{
    double l , r;
    bool operator < (const Seg& t) const
    {
        return r < t.r;
    }
}seg[N];
int main()
{
    cin >> n >> d;
    bool flag = false;
    for(int i=0;i<n;i++)
    {
        int x , y ;
        scanf("%d%d",&x,&y);
        if(y > d)
        {
             flag = true;
             break;
        }
        auto len = sqrt(d * d - y * y);
        seg[i] = {x - len , x + len};
    }

    if(flag)
    {
         puts("-1");
         return 0;
    }
    sort(seg,seg+n);
    int cnt = 0;
    double last = -1e20;    
    for(int i=0;i<n;i++)
    {
        if(last < seg[i].l)
        {
            last = seg[i].r;
            cnt++;
        }
    }
    cout << cnt << endl;
    return 0;
}
